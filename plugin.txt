function client_customization_add_message( $content ) {
    if ( is_singular( 'post' ) ) {
        $message = sprintf(
            '<p><b>%s: %s (%s)</b></p>',
            __( 'This content is created by', 'client-customization' ),
            get_bloginfo( 'name' ),
            get_bloginfo( 'url' )
        );
        $content .= $message;
    }

    return $content;
}
add_filter( 'the_content', 'client_customization_add_message', 10 );
